【水流动参数】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

【对流换热系数计算】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  雷诺数 Re = ρ × u × d / μ
  Re = 14.491421 × 2.00 × 0.034000 / 2.008024e-05
  Re = 49074
  
  努塞尔数 Nu = 0.023 × Re^0.8 × Pr^0.4
  Nu = 0.023 × 49074^0.8 × 1.052432^0.4
  Nu = 132.83

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  f = 0.005308
  
  压降 ΔP = f × (L/d) × (ρ × u² / 2)
  ΔP = 0.005308 × (94.37 / 0.034000) × (14.491421 × 2.00² / 2)

五、总传热系数计算
======================================================================

【各热阻计算】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  烟气侧热阻 R_gas = 1 / h_gas
  R_gas = 1 / 5.07
  R_gas = 1.973858e-01 (m²·K)/W
  
  管壁导热热阻 R_wall = (d_o / 2k) × ln(d_o/d_i)
  R_wall = (0.041000 / (2 × 50.00)) × ln(41.00 / 34.00)
  R_wall = 7.675673e-05 (m²·K)/W
  
  水侧热阻 R_water = d_o / (h_water × d_i)
  R_water = 0.041000 / (197.40 × 0.034000)
  R_water = 6.108701e-03 (m²·K)/W

【总传热系数】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  总热阻 R_total = R_gas + R_wall + R_water
  R_total = 1.973858e-01 + 7.675673e-05 + 6.108701e-03
  R_total = 2.035713e-01 (m²·K)/W
  
  总传热系数 K = 1 / R_total
  K = 1 / 2.035713e-01
  K = 4.91 W/(m²·K)

六、换热性能计算
======================================================================

【对数平均温差】
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  ΔT1 = T_gas_in - T_water_out
  ΔT1 = 800.00 - 540.46 = 259.54 °C
  
  ΔT2 = T_gas_out - T_water_in
  ΔT2 = 594.00 - 30.00 = 564.00 °C
  
  对数平均温差 ΔT_lm = (ΔT1 - ΔT2) / ln(ΔT1/ΔT2)
  ΔT_lm = (259.54 - 564.00) / ln(259.54 / 564.00)
  ΔT_lm = 661.66 °C

【实际换热量】

  Q = K × A × ΔT_lm
  Q = 4.91 × 12.15 × 661.66
  Q = 14646.43 W
  Q = 14.65 kW
